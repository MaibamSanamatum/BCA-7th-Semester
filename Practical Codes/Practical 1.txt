//insertion sort
#include <stdio.h>

int main() {
    int n, i, j, key;
    int comparisons = 0;  // Comparison counter

    printf("Enter number of elements: ");
    scanf("%d", &n);

    int arr[n];

    printf("Enter %d elements:\n", n);
    for(i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    // Insertion Sort Algorithm with comparison counter
    for(i = 1; i < n; i++) {
        key = arr[i];
        j = i - 1;

        while(j >= 0 && arr[j] > key) {
            comparisons++;      // Count successful comparison (arr[j] > key)
            arr[j + 1] = arr[j];
            j--;
        }
        
        if(j >= 0) {
            comparisons++;     // Count final failed comparison
        }

        arr[j + 1] = key;
    }

    printf("\nSorted Array:\n");
    for(i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
    
    printf("\nTotal Comparisons: %d\n", comparisons);  // Prints at very end

    return 0;
}
