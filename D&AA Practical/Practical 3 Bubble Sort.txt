Algorithm BubbleSort:
Input: n, array A[0..n-1]
Output: Sorted array A, total comparisons

comparisons ← 0

Read n
Create array A of size n
For i ← 0 to n-1 do
    Read A[i]
EndFor

For i ← 0 to n-2 do
    For j ← 0 to n-i-2 do
        comparisons ← comparisons + 1
        
        If A[j] > A[j+1] then
            temp ← A[j]
            A[j] ← A[j+1]
            A[j+1] ← temp
        EndIf
    EndFor
EndFor



Program Code:

//BUBBLE SORT
#include <stdio.h>

int comparisons = 0;  // global counter for comparisons

int main() {
    int arr[100], n, i, j, temp;
    printf("Enter number of elements: ");
    scanf("%d", &n);

    printf("Enter %d elements:\n", n);
    for (i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    // Bubble sort with comparison counting
    for (i = 0; i < n - 1; i++) {
        for (j = 0; j < n - i - 1; j++) {
            comparisons++;  // count each comparison
            
            if (arr[j] > arr[j + 1]) {
                // swap arr[j] and arr[j+1]
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }

    printf("Sorted array:\n");
    for (i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");

    printf("Total comparisons: %d\n", comparisons);

    return 0;
}
