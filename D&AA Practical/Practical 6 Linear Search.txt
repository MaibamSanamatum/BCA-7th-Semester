PROGRAM CODE:
//LINEAR SEARCH
#include <stdio.h>

int main() {
    int n, key;
    
    printf("Enter number of elements: ");
    scanf("%d", &n);
    
    int arr[n];
    printf("Enter %d elements:\n", n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }
    
    printf("Enter key to search: ");
    scanf("%d", &key);
    
    // Linear Search (no comparison counter)
    int found = 0;
    for (int i = 0; i < n; i++) {
        if (arr[i] == key) {
            printf("Element found at index %d\n", i);
            found = 1;
            break;
        }
    }
    
    if (!found) {
        printf("Element not found\n");
    }
    
    return 0;
}
