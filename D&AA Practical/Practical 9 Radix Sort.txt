//RADIX SORT
#include <stdio.h>

long long comparisons = 0;  // Global comparison counter

// Find the maximum number
int getMax(int a[], int n)
{
    int max = a[0];
    for(int i = 1; i < n; i++) {
        comparisons++;  // Count comparison
        if(a[i] > max)
            max = a[i];
    }
    return max;
}

// Counting sort according to digit
void countingSort(int a[], int n, int exp)
{
    int output[20];
    int count[10] = {0};

    // Count occurrences (no comparisons needed)
    for(int i = 0; i < n; i++)
        count[(a[i] / exp) % 10]++;

    // Cumulative count (no comparisons needed)
    for(int i = 1; i < 10; i++)
        count[i] += count[i - 1];

    // Build output array (stable, no comparisons needed)
    for(int i = n - 1; i >= 0; i--)
    {
        int digit = (a[i] / exp) % 10;
        output[count[digit] - 1] = a[i];
        count[digit]--;
    }

    // Copy back (no comparisons needed)
    for(int i = 0; i < n; i++)
        a[i] = output[i];
}

// Radix Sort
void radixSort(int a[], int n)
{
    int max = getMax(a, n);

    for(int exp = 1; max / exp > 0; exp *= 10) {
        comparisons++;  // Count exp loop comparison
        countingSort(a, n, exp);
    }
}

int main()
{
    int a[20], n;

    printf("Enter number of elements: ");
    scanf("%d", &n);

    printf("Enter elements:\n");
    for(int i = 0; i < n; i++)
        scanf("%d", &a[i]);

    radixSort(a, n);

    printf("Sorted array:\n");
    for(int i = 0; i < n; i++)
        printf("%d ", a[i]);
    printf("\n");

    printf("Total number of comparisons = %lld\n", comparisons);

    return 0;
}
